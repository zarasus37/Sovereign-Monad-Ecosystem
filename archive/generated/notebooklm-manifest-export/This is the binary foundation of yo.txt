class MonadReducer:
    """Reduces raw Boolean results into 8-bit hex signatures."""
    
    @staticmethod
    def to_hex(results: list[bool]) -> str:
        """Converts 8 bits to a Hexadecimal string (e.g. 0xB4)"""
        byte_val = 0
        for i, bit in enumerate(results):
            if bit:
                byte_val |= (1 << i)
        return hex(byte_val).upper()

    @staticmethod
    def evaluate_clarity(hex_val: str) -> float:
        """Calculates percentage of bits set to 1."""
        byte_val = int(hex_val, 16)
        return bin(byte_val).count('1') / 8.0
